// ═══════════════════════════════════════════════════════════════════════
// SEKHAR FINANCIAL SERVICES — Google Apps Script
//
// This script does TWO things:
//   1. POST  — receives user data and saves it to the correct Sheet tab
//   2. GET   — returns all rows from any tab back to the admin panel
//
// HOW TO DEPLOY:
//   1. Open your Google Sheet → Extensions → Apps Script
//   2. Delete the default code → paste this entire script → Ctrl+S
//   3. Click Deploy → New Deployment → Web App
//      Execute as: Me (your Google account)
//      Who has access: Anyone
//   4. Click Deploy → copy the Web App URL
//   5. Paste that URL into APP_CONFIG.SHEETS_WEBHOOK in the HTML file
// ═══════════════════════════════════════════════════════════════════════

// ── Entry point for all GET requests (admin reads data) ─────────────
function doGet(e) {
  try {
    const action = e.parameter.action || '';
    const sheet  = e.parameter.sheet  || '';

    if (action === 'read' && sheet) {
      const rows = readSheet(sheet);
      return jsonResponse({ status: 'ok', rows: rows });
    }

    return jsonResponse({ status: 'ok', message: 'SFS API ready. Use ?action=read&sheet=TabName' });
  } catch (err) {
    return jsonResponse({ status: 'error', message: err.message });
  }
}

// ── Entry point for all POST requests (users push data) ─────────────
function doPost(e) {
  try {
    const raw  = e.postData ? e.postData.contents : '{}';
    const data = JSON.parse(raw);
    const event = data.event || 'UNKNOWN';

    const handlers = {
      REGISTRATION:   writeRegistration,
      LOGIN:          writeLogin,
      ASSESSMENT:     writeAssessment,
      REPORT_PAYMENT: writePayment,
      SUBSCRIPTION:   writeSubscription,
    };

    if (handlers[event]) handlers[event](data);

    return jsonResponse({ status: 'ok', event: event });
  } catch (err) {
    Logger.log(err.toString());
    return jsonResponse({ status: 'error', message: err.message });
  }
}

// ── READ: return all rows from a tab as array of objects ────────────
function readSheet(tabName) {
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(tabName);
  if (!sheet || sheet.getLastRow() < 2) return [];

  const data    = sheet.getDataRange().getValues();
  const headers = data[0];
  const rows    = [];

  for (let i = 1; i < data.length; i++) {
    const obj = {};
    headers.forEach((h, j) => { obj[h] = data[i][j] !== undefined ? String(data[i][j]) : ''; });
    rows.push(obj);
  }
  return rows;
}

// ── WRITE handlers ──────────────────────────────────────────────────

function writeRegistration(d) {
  const s = getSheet('Registrations');
  ensureHeader(s, [
    'Timestamp', 'Name', 'Email', 'Mobile',
    'Referral Code', 'Plan', 'Source', 'Registered At', 'Password (b64)'
  ]);
  s.appendRow([
    d.timestamp, d.name, d.email, d.mobile || '',
    d.referral || '', d.plan || 'free', d.source || 'web',
    d.registered || d.timestamp, d.password_b64 || ''
  ]);
  autoFormat(s);
}

function writeLogin(d) {
  const s = getSheet('Logins');
  ensureHeader(s, ['Timestamp', 'Name', 'Email', 'Mobile', 'Plan', 'Login Time']);
  s.appendRow([d.timestamp, d.name, d.email, d.mobile || '', d.plan || '', d.lastLogin || d.timestamp]);
  autoFormat(s);
}

    'Home Loan Bal', 'Home EMI', 'Personal Loan', 'Personal EMI', 'Car Loan', 'Car EMI', 'Total EMI',
    'Life SA', 'Health Cover', 'Has Term Ins',
    'Emergency Fund', 'MF Corpus', 'EPF/PPF', 'FD/Bonds', 'Total Corpus', 'Monthly SIP', 'Risk Profile',
    'Monthly Surplus', 'Debt Ratio %', 'Emergency Months', 'Health Score', 'Assessments Done'
  ]);
  s.appendRow([
    d.timestamp, d.name, d.email, d.mobile||'', d.dob||'', d.plan||'free',
    d.monthly_salary||0, d.rental_income||0, d.other_income||0, d.annual_gross||0, d.spouse_income||0, d.total_income||0,
    d.household_exp||0, d.edu_exp||0, d.medical_exp||0, d.lifestyle_exp||0, d.insurance_prem||0, d.other_exp||0, d.total_expenses||0,
    d.home_loan_bal||0, d.home_emi||0, d.personal_loan||0, d.personal_emi||0, d.car_loan||0, d.car_emi||0, d.total_emi||0,

function writePayment(d) {
  const s = getSheet('Payments');
  ensureHeader(s, ['Timestamp', 'Name', 'Email', 'Mobile', 'Amount', 'UTR', 'UPI ID', 'Plan', 'Registered', 'Status']);
  s.appendRow([
    d.timestamp, d.name, d.email, d.mobile||'',
    d.amount||199, d.utr||'', d.upi_id||'', d.plan||'', d.registered||'',
    d.status||'pending_verification'
  ]);
  autoFormat(s);
}

function writeSubscription(d) {
  const s = getSheet('Subscriptions');
  ensureHeader(s, ['Timestamp', 'Name', 'Email', 'Mobile', 'Plan', 'Price', 'Method', 'Message', 'Status']);
  s.appendRow([
    d.timestamp, d.name, d.email, d.mobile||'',
    d.plan||'', d.price||'', d.method||'manual', d.message||'', d.status||'pending'
  ]);
  autoFormat(s);
}

// ── Helpers ─────────────────────────────────────────────────────────

function getSheet(name) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  return ss.getSheetByName(name) || ss.insertSheet(name);
}

function ensureHeader(sheet, cols) {
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(cols);
    const hdr = sheet.getRange(1, 1, 1, cols.length);
    hdr.setFontWeight('bold').setBackground('#0a1628').setFontColor('#c9a84c').setFontSize(11);
    sheet.setFrozenRows(1);
  }
}

function autoFormat(sheet) {
  if (sheet.getLastRow() > 1) {
    const row = sheet.getLastRow();
    sheet.getRange(row, 1, 1, sheet.getLastColumn())
         .setBackground(row % 2 === 0 ? '#f8f5ee' : '#ffffff');
  }
  try { sheet.autoResizeColumns(1, sheet.getLastColumn()); } catch(e) {}
}

function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
